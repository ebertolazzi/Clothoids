43 void
44 gfun( real_type alpha, real_type g[3] ) const {
45 real_type so = sin(alpha);
46 real_type co = cos(alpha);
47 real_type oco = alpha*co;
48 g[0] = so + oco;
49 g[1] = 2*co - alpha*so;
50 g[2] = -3*so - oco;
51 }
