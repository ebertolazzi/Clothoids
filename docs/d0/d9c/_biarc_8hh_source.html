50  gfun( real_type alpha, real_type g[3] ) const {
51  real_type so = sin(alpha);
52  real_type co = cos(alpha);
53  real_type oco = alpha*co;
54  g[0] = so + oco;
55  g[1] = 2*co - alpha*so;
56  g[2] = -3*so - oco;
57  }
